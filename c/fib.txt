public class InsertFibo{      
    public  void insertList(Integer n){
        List<Fibonacci_c> tmplist=new List<Fibonacci_c>();
        Integer next;
        Integer first=0;
        Integer second=1;
        for(Integer i=0;i<n;i++){
            Fibonacci_c obj=new Fibonacci_c();
                next=first+second;
                obj.Name=string.valueof(next);
                tmplist.add(obj);
                first=second;
                second=next;    
        }
        insert tmplist;
    }   
}

InsertFibo objfib = new InsertFibo();
objfib.display();
